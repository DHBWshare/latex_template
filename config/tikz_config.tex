\tikzmath{
    \winkel = 83;
    \hoehe = 1cm;
    \TanValue = {tan(\winkel)};
    \TanSecValue = {tan(7)};
    function oberseite(\a) {
        return (\a - 2*(\hoehe / \TanValue));
    };
%
    function position(\b){
        return \b + \TanSecValue * 2*\hoehe;
    };
    \unten = 2cm;
    \mitte = oberseite(\unten);
    \oben = oberseite(\mitte);
    \obenoben = oberseite(\oben);
%
    \pos1 = position(2*\obenoben);
    \pos2 = position(\pos1);
    \pos3 = position(\pos2);
}
